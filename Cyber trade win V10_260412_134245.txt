import json, sqlite3, os
from datetime import datetime, timedelta
from contextlib import contextmanager
from loguru import logger

MEMORY_DB = os.getenv("MEMORY_DB_PATH", "learn_memory.db")
SCHEMA = """
CREATE TABLE IF NOT EXISTS propostas (
    id INTEGER PRIMARY KEY, tipo TEXT, camada TEXT, titulo TEXT,
    problema TEXT, proposta_texto TEXT, parametro TEXT, valor_atual TEXT,
    valor_proposto TEXT, base_trades INTEGER, confianca_pct REAL,
    melhoria_pf_est REAL, risco TEXT, modo_paper INTEGER, fonte TEXT,
    dados_suporte TEXT, timestamp TEXT, status TEXT DEFAULT 'PENDENTE',
    decisao_ts TEXT, motivo_rejeicao TEXT, paper_trades INTEGER DEFAULT 0,
    paper_acerto REAL DEFAULT 0, impacto_real_pf REAL DEFAULT 0, rollback_ts TEXT
);
CREATE TABLE IF NOT EXISTS queries_pesquisa (id INTEGER PRIMARY KEY AUTOINCREMENT, query TEXT, timestamp TEXT);
CREATE TABLE IF NOT EXISTS ciclos_learn (id INTEGER PRIMARY KEY AUTOINCREMENT, timestamp TEXT, propostas_total INTEGER);
CREATE TABLE IF NOT EXISTS audit_trail (id INTEGER PRIMARY KEY AUTOINCREMENT, timestamp TEXT, acao TEXT, proposta_id INTEGER, detalhes TEXT);
"""

class LearnMemory:
    def __init__(self):
        with self._conn() as c: c.executescript(SCHEMA)
        self._id_counter = self._ultimo_id() + 1
    @contextmanager
    def _conn(self):
        conn=sqlite3.connect(MEMORY_DB); conn.row_factory=sqlite3.Row
        try: yield conn; conn.commit()
        finally: conn.close()
    def proximo_id(self)->int: i=self._id_counter; self._id_counter+=1; return i
    def _ultimo_id(self)->int:
        with self._conn() as c: r=c.execute("SELECT MAX(id) FROM propostas").fetchone(); return r[0] or 0
    def salvar_proposta(self, p)->int:
        with self._conn() as c:
            c.execute("INSERT OR REPLACE INTO propostas VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)",
                [p.id,p.tipo,p.camada,p.titulo,p.problema,p.proposta_texto,p.parametro,
                 json.dumps(p.valor_atual),json.dumps(p.valor_proposto),p.base_trades,p.confianca_pct,
                 p.melhoria_pf_est,p.risco,p.modo_paper,p.fonte,json.dumps(p.dados_suporte),
                 p.timestamp,p.status,None,None,0,0.0,0.0,None])
        return p.id
    def marcar_aprovada(self, pid:int): self._update_status(pid,"APROVADA_PAPER")
    def marcar_rejeitada(self, pid:int, motivo:str=""): self._update_status(pid,"REJEITADA",motivo)
    def registrar_rollback(self, pid:int): self._update_status(pid,"ROLLBACK")
    def _update_status(self, pid, status, motivo=""):
        with self._conn() as c:
            if motivo: c.execute("UPDATE propostas SET status=?,decisao_ts=?,motivo_rejeicao=? WHERE id=?",[status,datetime.now().isoformat(),motivo,pid])
            else: c.execute("UPDATE propostas SET status=?,decisao_ts=? WHERE id=?",[status,datetime.now().isoformat(),pid])
    def get_proposta(self, pid):
        with self._conn() as c: r=c.execute("SELECT * FROM propostas WHERE id=?",[pid]).fetchone(); return dict(r) if r else None
    def resumo_semana(self)->str: return "🧠 LEARN WIN — Resumo semanal disponível via interface."

import json, os, re
from datetime import datetime
from loguru import logger
from learn.learn_optimizer import Proposta
from infrastructure.telegram_bot import TelegramBot

OPENCLAW_JSON = os.getenv("OPENCLAW_JSON_PATH", "openclaw.json")

class LearnTelegramHandler:
    def __init__(self, learn_agent, tg:TelegramBot):
        self.learn=learn_agent; self.tg=tg; self.memory=learn_agent.memory
    async def enviar_proposta(self, p:Proposta):
        self.memory.salvar_proposta(p)
        icon="🧠" if p.camada=="OPTIMIZER" else "🔬"
        msg=f"{icon} LEARN WIN — Proposta #{p.id}\n━━━━━━━━━━━\n📊 PROBLEMA:\n{p.problema}\n\n💡 PROPOSTA:\n{p.proposta_texto}"
        if p.parametro: msg+=f"\n🔧 PARÂMETRO: {p.parametro}\n   Atual: {p.valor_atual} → Proposto: {p.valor_proposto}"
        msg+=f"\n\n📈 IMPACTO: Base {p.base_trades} trades | Confiança {p.confianca_pct:.0f}% | PF +{p.melhoria_pf_est:.2f} | Risco {p.risco}\n━━━━━━━━━━━\n✅ APROVO #{p.id} | ❌ REJEITO #{p.id} | 📊 DADOS #{p.id}"
        self.tg.alertar(msg)
    async def processar_resposta(self, mensagem:str):
        m=mensagem.upper().strip()
        if "APROVO" in m: await self._implementar_paper(self._extrair_id(m))
        elif "REJEITO" in m: self._rejeitar(self._extrair_id(m), mensagem)
        elif "DADOS" in m: self._enviar_dados(self._extrair_id(m))
        elif mensagem.lower().startswith("learn"): await self.learn.comando_manual(mensagem)
    async def _implementar_paper(self, pid:int):
        p=self.memory.get_proposta(pid)
        if not p: return self.tg.alertar("Proposta não encontrada.")
        self.memory.marcar_aprovada(pid)
        if p.get("parametro"): self._atualizar_parametro(p["parametro"], json.loads(p.get("valor_proposto")), pid)
        self.tg.alertar(f"✅ #{pid} aprovada para paper. Monitorando {p.get('modo_paper',10)} trades.")
    def _rejeitar(self, pid, motivo): self.memory.marcar_rejeitada(pid, motivo); self.tg.alertar(f"❌ #{pid} rejeitada.")
    def _enviar_dados(self, pid):
        p=self.memory.get_proposta(pid)
        if p: self.tg.alertar(f"📊 Dados #{pid}:\n{p.get('dados_suporte','{}')[:1500]}")
    def _atualizar_parametro(self, parametro, valor, pid):
        try:
            with open(OPENCLAW_JSON,"r") as f: config=json.load(f)
            ref=config
            for parte in parametro.split(".")[:-1]:
                if parte not in ref: ref[parte]={}
                ref=ref[parte]
            ref[parametro.split(".")[-1]]=valor
            config.setdefault("learn_changes",[]).append({"proposta_id":pid,"parametro":parametro,"valor_novo":valor,"timestamp":datetime.now().isoformat()})
            with open(OPENCLAW_JSON,"w") as f: json.dump(config,f,indent=2)
        except Exception as e: logger.error(f"Falha update: {e}")
    @staticmethod
    def _extrair_id(texto):
        nums=re.findall(r'#(\d+)',texto.upper())
        return int(nums[0]) if nums else None

  risco_financeiro = capital × 0.01
  risco_contrato = distancia_stop_pts × 0.20
  contratos = min(floor(risco_financeiro / risco_contrato), 5)
